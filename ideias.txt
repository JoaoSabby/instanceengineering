// [[Rcpp::export]]
arma::mat GenerateSyntheticAdasynCpp(
  const arma::mat& minorityMatrix, 
  const arma::umat& minorityNeighborIndex, 
  const arma::uvec& syntheticPerRow) {
  
  int syntheticCount = arma::sum(syntheticPerRow);
  int colCount = minorityMatrix.n_cols;
  arma::mat syntheticMatrix(syntheticCount, colCount);
  
  int writeStart = 0;
  for(size_t i = 0; i < syntheticPerRow.n_elem; ++i) {
    int rowCount = syntheticPerRow[i];
    if(rowCount > 0) {
      arma::mat baseRows = arma::repmat(minorityMatrix.row(i), rowCount, 1);
      
      // Sorteia vizinhos aleatórios para a interpolação
      arma::uvec randomIndices = arma::randi<arma::uvec>(rowCount, arma::distr_param(0, minorityNeighborIndex.n_cols - 1));
      arma::uvec selectedNeighborRows = minorityNeighborIndex.row(i).elem(randomIndices);
      
      arma::mat neighborRows = minorityMatrix.rows(selectedNeighborRows);
      arma::mat randomWeights = arma::randu<arma::mat>(rowCount, 1);
      
      // Expande os pesos aleatórios para todas as colunas
      arma::mat expandedWeights = arma::repmat(randomWeights, 1, colCount);
      
      syntheticMatrix.rows(writeStart, writeStart + rowCount - 1) = baseRows + expandedWeights % (neighborRows - baseRows);
      writeStart += rowCount;
    }
  }
  return syntheticMatrix;
}

ApplyZScoreScalingMatrix <- function(xMatrix, scalingInfo){
  xMatrix <- OverUnderAsNumericMatrix(xMatrix)
  ValidateScalingInfo(scalingInfo, NCOL(xMatrix))
  
  # A transposição dupla explora a reciclagem de vetores do R em C
  scaled <- t((t(xMatrix) - scalingInfo$centers) / scalingInfo$scales)
  storage.mode(scaled) <- "double"
  scaled
}

InferNumericColumnTypes <- function(dataFrame){
  columnNames <- OverUnderGetColumnNames(dataFrame)
  
  inferOne <- function(columnData){
    # Se já for inteiro nativo, retorna imediatamente
    if(is.integer(columnData)) return("integer")
    
    uniqueValues <- sort(unique(stats::na.omit(columnData)))
    if(length(uniqueValues) <= 2L && all(uniqueValues %in% c(0, 1))){
      return("binary")
    }
    
    # Verifica apenas os primeiros 1000 registros para economizar tempo
    sampleData <- head(columnData, 1000L)
    isIntegerLike <- all(abs(sampleData - round(sampleData)) < sqrt(.Machine$double.eps))
    if(isIntegerLike){
      return("integer")
    }
    "double"
  }
  
  # Aplica iterando nas colunas do dataframe, não da matriz
  inferredTypes <- vapply(dataFrame, inferOne, character(1L), USE.NAMES = FALSE)
  
  data.frame(
    columnName = columnNames,
    inferredType = inferredTypes,
    stringsAsFactors = FALSE
  )
}

// Função exportada para o R
SEXP GenerateSyntheticAdasynC(SEXP minorityMatrix, SEXP minorityNeighborIndex, SEXP syntheticPerRow) {
    
    // Extração das dimensões da matriz original
    int numMinorityRows = nrows(minorityMatrix);
    int colCount = ncols(minorityMatrix);
    int numNeighbors = ncols(minorityNeighborIndex);

    // Ponteiros diretos para os vetores em memória (column-major)
    double *pMinority = REAL(minorityMatrix);
    int *pNeighbor = INTEGER(minorityNeighborIndex);
    int *pSynPerRow = INTEGER(syntheticPerRow);

    // Pré-cálculo do total de linhas sintéticas para alocação
    int totalSyntheticCount = 0;
    int numGroups = length(syntheticPerRow);
    for(int i = 0; i < numGroups; i++) {
        totalSyntheticCount += pSynPerRow[i];
    }

    // Alocação da matriz de saída e proteção contra o Garbage Collector
    SEXP syntheticMatrix = PROTECT(allocMatrix(REALSXP, totalSyntheticCount, colCount));
    double *pOutput = REAL(syntheticMatrix);

    // Inicialização do estado RNG nativo do R
    GetRNGstate(); 

    int writeStart = 0;
    for(int i = 0; i < numGroups; i++) {
        int rowCount = pSynPerRow[i];
        
        if(rowCount > 0) {
            for(int r = 0; r < rowCount; r++) {
                // Sorteio de um vizinho aleatório (unif_rand gera [0, 1))
                int randIdx = (int)(unif_rand() * numNeighbors);
                
                // Extração do índice do vizinho
                // O R passa índices base 1 (1-based), o C usa base 0 (0-based)
                // A indexação column-major: row + col * numRows
                int neighborIndexPos = i + randIdx * numMinorityRows;
                int selectedNeighborRow = pNeighbor[neighborIndexPos] - 1;

                // Peso da interpolação linear
                double weight = unif_rand();

                // Escrita vetorizada manual por coluna
                int outRow = writeStart + r;
                for(int j = 0; j < colCount; j++) {
                    double baseVal = pMinority[i + j * numMinorityRows];
                    double neighborVal = pMinority[selectedNeighborRow + j * numMinorityRows];
                    
                    // Cálculo e inserção na matriz de saída (column-major)
                    pOutput[outRow + j * totalSyntheticCount] = baseVal + weight * (neighborVal - baseVal);
                }
            }
            writeStart += rowCount;
        }
    }

    // Devolução do estado RNG ao R e libertação da proteção de memória
    PutRNGstate();
    UNPROTECT(1);
    
    return syntheticMatrix;
}

#' Gerar amostras sintéticas através da C API
#' @noRd
GenerateAdasynSamplesFast <- function(
  minorityMatrix, 
  neighborIndexMatrix, 
  syntheticPerRowVector
){
  # Garantia estrita de tipos para evitar falhas de segmentação no C
  storage.mode(minorityMatrix) <- "double"
  storage.mode(neighborIndexMatrix) <- "integer"
  syntheticPerRowInteger <- as.integer(syntheticPerRowVector)
  
  syntheticMatrix <- .Call(
    "GenerateSyntheticAdasynC", 
    minorityMatrix, 
    neighborIndexMatrix, 
    syntheticPerRowInteger,
    PACKAGE = "adanear"
  )
  
  colnames(syntheticMatrix) <- colnames(minorityMatrix)
  
  syntheticMatrix
}

#' Calcular parametros de z-score
#' @noRd
ComputeZScoreParams <- function(xMatrix){
  xMatrix <- OverUnderAsNumericMatrix(xMatrix)
  
  # Rfast calcula medias e desvios padrao vetorialmente
  centersVector <- Rfast::colmeans(xMatrix)
  scalesVector <- Rfast::colVars(xMatrix, std = TRUE)

  invalidMask <- is.na(scalesVector) | !is.finite(scalesVector) | scalesVector <= 0
  if(any(invalidMask)){
    columnNames <- colnames(xMatrix)
    if(is.null(columnNames)){
      columnNames <- paste0("V", seq_len(NCOL(xMatrix)))
    }
    stop(
      paste0(
        "Colunas com desvio padrao zero ou indefinido: ",
        paste(columnNames[invalidMask], collapse = ", ")
      ),
      call. = FALSE
    )
  }

  list(centers = centersVector, scales = scalesVector)
}

#' Aplicar z-score em matrix double
#' @noRd
ApplyZScoreScalingMatrix <- function(xMatrix, scalingInfo){
  xMatrix <- OverUnderAsNumericMatrix(xMatrix)
  ValidateScalingInfo(scalingInfo, NCOL(xMatrix))
  
  # eachrow substitui o sweep com performance superior
  centeredMatrix <- Rfast::eachrow(xMatrix, scalingInfo$centers, oper = "-")
  scaledMatrix <- Rfast::eachrow(centeredMatrix, scalingInfo$scales, oper = "/")
  
  storage.mode(scaledMatrix) <- "double"
  scaledMatrix
}

#' Reverter z-score em matrix double
#' @noRd
RevertZScoreScalingMatrix <- function(xMatrix, scalingInfo){
  xMatrix <- OverUnderAsNumericMatrix(xMatrix)
  ValidateScalingInfo(scalingInfo, NCOL(xMatrix))
  
  # Multiplica pela escala e soma o centro
  unscaledMatrix <- Rfast::eachrow(xMatrix, scalingInfo$scales, oper = "*")
  restoredMatrix <- Rfast::eachrow(unscaledMatrix, scalingInfo$centers, oper = "+")
  
  storage.mode(restoredMatrix) <- "double"
  restoredMatrix
}

#' Carregar dependencias do fluxo de sampling
#' @noRd
OverUnderLoadPackages <- function(){
  if(isTRUE(.OverUnderState$packagesLoaded)){
    return(invisible(TRUE))
  }

  packageNamesVector <- c("FNN", "Rfast")
  for(packageName in packageNamesVector){
    if(!requireNamespace(packageName, quietly = TRUE)){
      stop("Pacote necessario nao encontrado: ", packageName, call. = FALSE)
    }
  }

  .OverUnderState$packagesLoaded <- TRUE
  invisible(TRUE)
}
